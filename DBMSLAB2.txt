**Schema (MySQL v5.7)**

    CREATE TABLE SALESMAN(SALESMAN_ID INTEGER, NAME VARCHAR(20), CITY VARCHAR(20), COMMISSION VARCHAR(20), PRIMARY KEY (SALESMAN_ID));
    CREATE TABLE CUSTOMER(CUSTOMER_ID INTEGER, CUSTOMER_NAME VARCHAR(20), CITY VARCHAR(20), GRADE INTEGER, SALESMAN_ID INTEGER, PRIMARY KEY (CUSTOMER_ID), FOREIGN KEY(SALESMAN_ID) REFERENCES SALESMAN(SALESMAN_ID) ON DELETE SET NULL);
    CREATE TABLE ORDERS(ORD_NO INTEGER, PURCHASE_AMT INTEGER, ORD_DATE DATE,SALESMAN_ID INTEGER,CUSTOMER_ID INTEGER, PRIMARY KEY (ORD_NO),FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID) ON DELETE CASCADE,FOREIGN KEY(SALESMAN_ID) REFERENCES SALESMAN(SALESMAN_ID) ON DELETE CASCADE);
